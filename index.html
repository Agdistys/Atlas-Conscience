#!/usr/bin/env python3 # -*- coding: utf-8 -*- from __future__ import annotations import json import re import sys import unicodedata from pathlib import Path from pyvis.network import Network from atlas_reasoner import AtlasReasoner, find_data_dir DATA_DIR = Path("./data") PAIRS_PATH = DATA_DIR / "pairs.json" HUBS_PATH = DATA_DIR / "hubs.json" TRIADS_PATH = DATA_DIR / "triads.json" CONCEPT_NAV_PATH = DATA_DIR / "concept_pages.json" OUTPUT_PATH = Path("atlas_map.html") MAX_EDGES = 450 MIN_EDGE_WEIGHT = 2 TOP_TRIADS_TO_DRAW = 18 COSMOLOGY = { "VIVANT": ["conscience", "vivant", "sens", "nature", "corps", "monde", "vie"], "LOI": ["charte", "verite", "justice", "coherence", "respect", "droit", "lois"], "OMBRE": ["violence", "deni", "biais", "systeme", "peur", "domination", "maladie"], "TEMPLE": ["art", "nature", "corps", "vivant", "presence", "beaute", "onde"], "VERBE": ["verbe", "esprit", "partager", "ensemble", "agdistys", "symbolique"], } CLUSTER_COLORS = { "VIVANT": "#F4F1EA", "LOI": "#D4AF37", "OMBRE": "#D1495B", "TEMPLE": "#33D1C6", "VERBE": "#E13BFF", "MIXTE": "#8A8F98", } def normalize_text(text: str) -> str: text = text.lower() text = text.replace("œ", "oe").replace("æ", "ae") text = unicodedata.normalize("NFKD", text) text = "".join(ch for ch in text if not unicodedata.combining(ch)) text = re.sub(r"[’']", " ", text) text = re.sub(r"[^a-zA-Z0-9_\- ]+", " ", text) text = re.sub(r"\s+", " ", text).strip() return text def safe_load_json(path: Path, default): if not path.exists(): return default try: with path.open("r", encoding="utf-8") as f: return json.load(f) except Exception as e: print(f"[warn] impossible de lire {path}: {e}") return default def load_pairs() -> list[dict]: pairs = safe_load_json(PAIRS_PATH, []) clean = [] for item in pairs: if not isinstance(item, dict): continue a = normalize_text(item.get("a", "")) b = normalize_text(item.get("b", "")) try: weight = float(item.get("weight", 1)) except Exception: weight = 1.0 if not a or not b or a == b: continue if weight < MIN_EDGE_WEIGHT: continue clean.append({"a": a, "b": b, "weight": weight}) clean.sort(key=lambda x: x["weight"], reverse=True) return clean[:MAX_EDGES] def load_hubs() -> dict[str, dict]: hubs = safe_load_json(HUBS_PATH, []) result = {} for item in hubs: if not isinstance(item, dict): continue node = normalize_text(item.get("node", "")) if not node: continue result[node] = { "degree": item.get("degree", 1), "weighted_degree": float(item.get("weighted_degree", 1)), } return result def load_triads() -> list[dict]: triads = safe_load_json(TRIADS_PATH, []) clean = [] for item in triads: if not isinstance(item, dict): continue a = normalize_text(item.get("a", "")) b = normalize_text(item.get("b", "")) c = normalize_text(item.get("c", "")) try: score = float(item.get("score", 0)) except Exception: score = 0.0 if not a or not b or not c: continue clean.append({"a": a, "b": b, "c": c, "score": score}) clean.sort(key=lambda x: x["score"], reverse=True) return clean def load_concept_nav() -> dict[str, list[str]]: raw = safe_load_json(CONCEPT_NAV_PATH, {}) clean = {} if not isinstance(raw, dict): return clean for k, v in raw.items(): nk = normalize_text(str(k)) if isinstance(v, list): clean[nk] = [str(x) for x in v[:5]] return clean def detect_cluster(node: str) -> str: memberships = [cluster for cluster, words in COSMOLOGY.items() if node in words] if len(memberships) == 1: return memberships[0] if len(memberships) > 1: return "MIXTE" return "MIXTE" def compute_node_size(node: str, hubs: dict[str, dict], activated_nodes: set[str]) -> float: if node not in hubs: base = 12 else: weighted = hubs[node]["weighted_degree"] if weighted >= 250: base = 50 elif weighted >= 150: base = 38 elif weighted >= 100: base = 29 elif weighted >= 70: base = 21 elif weighted >= 50: base = 17 else: base = 13 if node in activated_nodes: base += 8 return base def get_font_color(cluster: str) -> str: return "#111111" if cluster == "VIVANT" else "#FFFFFF" def parse_query(argv: list[str]) -> str | None: if len(argv) >= 3 and argv[1] == "--query": return " ".join(argv[2:]).strip() return None def get_query_activation(query: str | None): if not query: return set(), set(), [] reasoner = AtlasReasoner(find_data_dir()) result = reasoner.answer(query) activated_nodes = { normalize_text(item["concept"]) for item in result.get("activated_concepts", []) } activated_edges = set() for link in result.get("links", []): if not link.get("internal"): continue a = normalize_text(link["a"]) b = normalize_text(link["b"]) activated_edges.add(tuple(sorted((a, b)))) return activated_nodes, activated_edges, result.get("activated_concepts", []) def get_top_triads_for_query( triads: list[dict], activated_nodes: set[str], top_n: int = TOP_TRIADS_TO_DRAW ): if not activated_nodes: return triads[:top_n] scored = [] for t in triads: triad_nodes = {t["a"], t["b"], t["c"]} overlap = len(triad_nodes.intersection(activated_nodes)) if overlap == 0: continue scored.append({ **t, "overlap": overlap, "boosted_score": t["score"] + overlap * 20, }) scored.sort(key=lambda x: (x["overlap"], x["boosted_score"]), reverse=True) return scored[:top_n] def build_overlay_html(query: str | None) -> str: query_block = "" if query: query_block = f"""
Question active
{query}
""" return f"""
Atlas de ConScience
Planétarium cognitif interactif issu du graphe Ouroboros.
● VIVANT
● LOI
● OMBRE / MALADIE
● TEMPLE
● VERBE / SYMBOLIQUE
● MIXTE
Plus un nœud est grand, plus il est hub.
Bord blanc = activé par la question.
Liens plus clairs = activés ou triades mises en avant.
{query_block}
Lire la note explicative
Nœud sélectionné
Clique sur un concept pour voir ses informations et les pages liées.
""" def inject_ui(html_path: Path, query: str | None, concept_nav: dict[str, list[str]]): html = html_path.read_text(encoding="utf-8") concept_nav_json = json.dumps(concept_nav, ensure_ascii=False) script = f""" """ block = build_overlay_html(query) if "" in html: html = html.replace("", block + "\n" + script + "\n") html_path.write_text(html, encoding="utf-8") def build_interactive_map(query: str | None = None): pairs = load_pairs() hubs = load_hubs() triads = load_triads() concept_nav = load_concept_nav() if not pairs: print("[error] aucun pair exploitable trouvé dans data/pairs.json") return activated_nodes, activated_edges, _ = get_query_activation(query) highlighted_triads = get_top_triads_for_query(triads, activated_nodes) triad_edges = set() for t in highlighted_triads: triad_edges.add(tuple(sorted((t["a"], t["b"])))) triad_edges.add(tuple(sorted((t["a"], t["c"])))) triad_edges.add(tuple(sorted((t["b"], t["c"])))) net = Network( height="930px", width="100%", bgcolor="#050A12", font_color="white", notebook=False, directed=False, ) added_nodes = set() for pair in pairs: a = pair["a"] b = pair["b"] weight = pair["weight"] edge_key = tuple(sorted((a, b))) for node in (a, b): if node in added_nodes: continue cluster = detect_cluster(node) color = CLUSTER_COLORS.get(cluster, CLUSTER_COLORS["MIXTE"]) size = compute_node_size(node, hubs, activated_nodes) font_color = get_font_color(cluster) is_activated = node in activated_nodes border_color = "#FFFFFF" if is_activated or size >= 34 else "rgba(255,255,255,0.24)" border_width = 3 if is_activated or size >= 34 else 1 title = ( f"{node}\n" f"Cluster: {cluster}\n" f"Weighted degree: {hubs.get(node, {}).get('weighted_degree', 'n/a')}\n" f"Degree: {hubs.get(node, {}).get('degree', 'n/a')}" ) if is_activated: title += "\nActivé par la question" net.add_node( node, label=node, title=title, color={ "background": color, "border": border_color, "highlight": {"background": color, "border": "#FFFFFF"}, "hover": {"background": color, "border": "#FFFFFF"}, }, size=size, font={"color": font_color, "size": 16 if size < 30 else 20, "face": "Arial"}, borderWidth=border_width, ) added_nodes.add(node) edge_width = max(1, min(9, weight / 2.0)) edge_color = "rgba(220,225,235,0.16)" if edge_key in triad_edges: edge_color = "rgba(255,255,255,0.28)" edge_width += 1 if edge_key in activated_edges: edge_color = "rgba(255,255,255,0.70)" edge_width += 2 net.add_edge( a, b, value=weight, width=edge_width, title=f"{a} ↔ {b} | poids {weight}", color=edge_color, ) net.heading = "" net.set_options(""" const options = { "nodes": { "shape": "dot", "scaling": { "min": 10, "max": 54 } }, "edges": { "smooth": { "enabled": true, "type": "dynamic" } }, "physics": { "enabled": true, "barnesHut": { "gravitationalConstant": -9800, "centralGravity": 0.07, "springLength": 150, "springConstant": 0.017, "damping": 0.12, "avoidOverlap": 0.20 }, "stabilization": { "enabled": true, "iterations": 340 } }, "interaction": { "hover": true, "navigationButtons": true, "keyboard": true, "tooltipDelay": 80 } } """) net.write_html(str(OUTPUT_PATH), open_browser=False) inject_ui(OUTPUT_PATH, query, concept_nav) print(f"[ok] carte interactive sauvegardée dans {OUTPUT_PATH}") if query: print(f"[info] question active : {query}") print("[info] ouvre atlas_map.html dans ton navigateur") if __name__ == "__main__": query = parse_query(sys.argv) build_interactive_map(query)
